BEGIN;

CREATE TABLE IF NOT EXISTS story_slides (
    id          BIGSERIAL   PRIMARY KEY,
    master_id   INTEGER     NOT NULL REFERENCES master(id) ON DELETE CASCADE,
    media_url   TEXT        NOT NULL,
    sort_order  INTEGER     NOT NULL DEFAULT 0,
    is_active   BOOLEAN     NOT NULL DEFAULT TRUE,
    created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

    ON story_slides (master_id, is_active, sort_order, id);

INSERT INTO story_slides (master_id, media_url, sort_order, is_active)
SELECT m.id, m.story_media_url, 0, TRUE
FROM master m
WHERE m.story_media_url IS NOT NULL AND m.story_media_url <> ''
  AND NOT EXISTS (SELECT 1 FROM story_slides s WHERE s.master_id = m.id);

'use client';

import { useState, useEffect, useRef, useCallback } from 'react';
import Image from 'next/image';
import Link from 'next/link';
import { X, ExternalLink, Lock, Copy, Check, Volume2, VolumeX } from 'lucide-react';
import type { Store } from '@/lib/api';
import { trackAction, logStoryView } from '@/lib/api';
import { useLang } from '@/lib/lang';
import { useAuthGate } from '@/lib/auth-gate';
import { useTrend } from '@/lib/trend';
import { logoVariant } from '@/lib/cloudinary';
import ReportCodeButton from '@/components/ReportCodeButton';

const STORY_DURATION_MS = 6000;
const isVideoUrl = (u: string | null | undefined) =>
  !!u && /\.(mp4|webm|mov|m4v)(\?|$)/i.test(u);

interface Props {
  stores: Store[];
  limit?: number;
}

export default function StoreStories({ stores, limit = 12 }: Props) {
  const { t, pick, isRTL } = useLang();
  const { isLoggedIn, promptAuth } = useAuthGate();
  const { trendKind } = useTrend();

  const featured = stores
    .filter((s) => s.is_promoted && s.affiliate_link)
    .slice(0, limit);

  const [openIndex, setOpenIndex] = useState<number | null>(null);   // أي متجر
  const [slideIndex, setSlideIndex] = useState(0);                   // أي شريحة داخله
  const [progress, setProgress] = useState(0);
  const [paused, setPaused] = useState(false);
  const [copied, setCopied] = useState(false);
  const [muted, setMuted] = useState(false);   // الصوت مُشغّل افتراضياً، مع زر تبديل
  const rafRef = useRef<number | null>(null);
  const startRef = useRef<number>(0);
  const videoRef = useRef<HTMLVideoElement | null>(null);
  // story_view_id: يولَّد عند فتح كل متجر؛ يُمرَّر للنسخ/النقر لربطه في التحليلات.
  const viewIdRef = useRef<string | null>(null);

  const current = openIndex !== null ? featured[openIndex] : null;
  const slides = current?.story_slides ?? [];
  const slideCount = Math.max(slides.length, 1);
  const currentMedia = slides.length > 0 ? slides[Math.min(slideIndex, slides.length - 1)] : null;
  const currentIsVideo = isVideoUrl(currentMedia);

  const close = useCallback(() => {
    setOpenIndex(null);
    setSlideIndex(0);
    setProgress(0);
    setPaused(false);
    setCopied(false);
    viewIdRef.current = null;
  }, []);

  // فتح متجر → شريحة 0 + تسجيل مشاهدة (للمسجّلين فقط)
  const openStore = useCallback((i: number) => {
    if (i < 0 || i >= featured.length) return;
    setOpenIndex(i);
    setSlideIndex(0);
    setProgress(0);
    setCopied(false);
    const store = featured[i];
    viewIdRef.current = store ? logStoryView(store.store_id) : null;
  }, [featured]);

  // التالي: شريحة تالية داخل المتجر، وإلا المتجر التالي، وإلا إغلاق
  const advance = useCallback(() => {
    if (openIndex === null) return;
    const cnt = Math.max((featured[openIndex]?.story_slides ?? []).length, 1);
    if (slideIndex < cnt - 1) {
      setSlideIndex((s) => s + 1);
      setProgress(0);
      setCopied(false);
    } else if (openIndex < featured.length - 1) {
      openStore(openIndex + 1);
    } else {
      close();
    }
  }, [openIndex, slideIndex, featured, openStore, close]);

  // السابق: شريحة سابقة، وإلا المتجر السابق (من بدايته)
  const back = useCallback(() => {
    if (openIndex === null) return;
    if (slideIndex > 0) {
      setSlideIndex((s) => s - 1);
      setProgress(0);
    } else if (openIndex > 0) {
      openStore(openIndex - 1);
    }
  }, [openIndex, slideIndex, openStore]);

  // مؤقّت الصورة/الشعار (6 ثوانٍ). الفيديو يقود تقدّمه بنفسه عبر onTimeUpdate/onEnded.
  useEffect(() => {
    if (openIndex === null || paused || currentIsVideo) return;
    startRef.current = performance.now() - progress * STORY_DURATION_MS;
    const tick = (now: number) => {
      const pct = Math.min((now - startRef.current) / STORY_DURATION_MS, 1);
      setProgress(pct);
      if (pct >= 1) advance();
      else rafRef.current = requestAnimationFrame(tick);
    };
    rafRef.current = requestAnimationFrame(tick);
    return () => {
      if (rafRef.current) cancelAnimationFrame(rafRef.current);
    };
    // progress عمداً خارج الـ deps حتى لا يُعاد تشغيل الـ RAF كل إطار.
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [openIndex, slideIndex, paused, currentIsVideo, advance]);

  // تشغيل/إيقاف الفيديو مع الضغط المطوّل + إعادة التشغيل عند تغيّر الشريحة.
  // لو منع المتصفّح التشغيل بصوت، نكتم ونعيد (حتى لا يتجمّد).
  useEffect(() => {
    const v = videoRef.current;
    if (!v) return;
    if (paused) { v.pause(); return; }
    v.play().catch(() => {
      if (!v.muted) { v.muted = true; setMuted(true); v.play().catch(() => {}); }
    });
  }, [paused, openIndex, slideIndex]);

  useEffect(() => {
    if (openIndex === null) return;
    const prevOverflow = document.body.style.overflow;
    document.body.style.overflow = 'hidden';
    const onKey = (e: KeyboardEvent) => {
      if (e.key === 'Escape') close();
      else if (e.key === 'ArrowRight') { isRTL ? back() : advance(); }
      else if (e.key === 'ArrowLeft')  { isRTL ? advance() : back(); }
    };
    document.addEventListener('keydown', onKey);
    return () => {
      document.removeEventListener('keydown', onKey);
      document.body.style.overflow = prevOverflow;
    };
  }, [openIndex, close, advance, back, isRTL]);

  if (featured.length === 0) return null;

  const currentName  = current ? pick(current.store_id, current.name_en) : '';
  const currentOffer = current ? (current.discount_value || pick(current.extra_offer, current.extra_offer_en)) : null;

  // ── كتلة الـ CTA المشتركة (الكوبون + الزيارة + الإبلاغ) ──
  const cta = current && (
    <>
      {current.public_coupon && (
        isLoggedIn ? (
          <div className="w-full max-w-xs mb-4">
            <p className="text-[10px] text-white/60 tracking-[0.3em] uppercase mb-2">
              {t('story_code_label')}
            </p>
            <button
              type="button"
              onClick={(e) => {
                e.stopPropagation();
                if (!current.public_coupon) return;
                navigator.clipboard?.writeText(current.public_coupon).catch(() => {});
                trackAction({
                  storeId: current.store_id,
                  action: 'copy_coupon',
                  storyViewId: viewIdRef.current || undefined,
                });
                setCopied(true);
                setPaused(true);
                setTimeout(() => setCopied(false), 1600);
              }}
              className="pointer-events-auto w-full px-4 py-3 rounded-xl bg-white/15
                         ring-1 ring-white/25 hover:bg-white/25 transition backdrop-blur-sm
                         font-mono text-lg font-bold tracking-widest text-white
                         flex items-center justify-center gap-2"
            >
              {copied
                ? <><Check className="w-4 h-4" /><span className="text-sm">{t('store_copied')}</span></>
                : <><span>{current.public_coupon}</span><Copy className="w-4 h-4 opacity-70" /></>}
            </button>
          </div>
        ) : (
          <button
            type="button"
            onClick={(e) => { e.stopPropagation(); promptAuth(); }}
            className="pointer-events-auto w-full max-w-xs mb-4 inline-flex
                       items-center justify-center gap-2 px-6 py-3.5 rounded-xl
                       bg-gradient-to-l from-amber-500 to-orange-500 text-white
                       font-bold hover:shadow-lg hover:shadow-amber-500/30 transition"
          >
            <Lock className="w-4 h-4" />
            <span>{t('sign_to_see_code')}</span>
          </button>
        )
      )}

      <Link
        href={`/store/${encodeURIComponent(current.store_id)}`}
        onClick={(e) => {
          e.stopPropagation();
          trackAction({
            storeId: current.store_id,
            action: 'click_link',
            storyViewId: viewIdRef.current || undefined,
          });
          close();
        }}
        className="pointer-events-auto w-full max-w-xs inline-flex items-center
                   justify-center gap-2 px-6 py-3.5 rounded-xl
                   bg-white text-slate-900 font-semibold
                   hover:bg-white/90 transition-colors"
      >
        <ExternalLink className="w-4 h-4" />
        <span>{t('story_visit')}</span>
      </Link>

      {current.public_coupon && isLoggedIn && (
        <div className="mt-3 pointer-events-auto">
          <ReportCodeButton
            storeId={current.store_id}
            variant="story"
            onOpen={() => setPaused(true)}
          />
        </div>
      )}
    </>
  );

  return (
    <>
      <section
        aria-label={t('story_aria')}
        className="w-full px-4 sm:px-6 lg:px-8 pt-6"
      >
        <div className="mx-auto max-w-7xl">
          <div
            className="flex gap-5 sm:gap-6 overflow-x-auto pb-3 px-1
                       [scrollbar-width:none] [&::-webkit-scrollbar]:hidden"
          >
            {featured.map((store, i) => {
              const name = pick(store.store_id, store.name_en);
              const isTrend = !!trendKind(store.store_id);
              return (
                <button
                  key={store.id}
                  onClick={() => openStore(i)}
                  className="story-item group flex flex-col items-center gap-2
                             flex-shrink-0 focus:outline-none
                             focus-visible:ring-2 focus-visible:ring-slate-400 rounded-2xl"
                  aria-label={`${t('story_open')} ${name}`}
                >
                  <span className={`story-ring story-ring-pulse relative grid place-items-center
                                   w-16 h-16 sm:w-[72px] sm:h-[72px] rounded-full${isTrend ? ' story-ring-trend' : ''}`}>
                    {isTrend && (
                      <span className={`absolute -top-1 z-10 text-sm drop-shadow ${isRTL ? '-right-1' : '-left-1'}`}>
                        🔥
                      </span>
                    )}
                    <span className="relative w-[calc(100%-8px)] h-[calc(100%-8px)]
                                     rounded-full overflow-hidden
                                     bg-white dark:bg-slate-900
                                     ring-2 ring-white dark:ring-slate-950">
                      <Image
                        src={logoVariant(store.logo_url, 72, 72)}
                        alt={name}
                        fill
                        sizes="72px"
                        unoptimized
                        className="object-contain p-2 transition-transform duration-500
                                   group-hover:scale-110"
                      />
                    </span>
                  </span>
                  <span className="text-[11px] sm:text-xs font-medium
                                   text-slate-700 dark:text-slate-300
                                   max-w-[72px] truncate">
                    {name}
                  </span>
                </button>
              );
            })}
          </div>
        </div>
      </section>

      {current && (
        <div
          role="dialog"
          aria-modal="true"
          aria-label={`${t('story_open')} ${currentName}`}
          className="fixed inset-0 z-50 bg-slate-950/85 backdrop-blur-md
                     flex items-center justify-center p-0 sm:p-6
                     animate-[storyFade_180ms_ease-out]"
          onClick={close}
        >
          <div
            className="story-card relative w-full h-full
                       sm:w-[420px] sm:h-[680px] sm:max-h-[90vh]
                       sm:rounded-3xl overflow-hidden
                       bg-gradient-to-br from-slate-900 via-slate-800 to-slate-900
                       shadow-2xl"
            onClick={(e) => e.stopPropagation()}
            onPointerDown={() => setPaused(true)}
            onPointerUp={() => setPaused(false)}
            onPointerLeave={() => setPaused(false)}
          >
            {/* ── الوسائط/الشعار (طبقة الخلفية ملء البطاقة) ── */}
            {currentMedia ? (
              currentIsVideo ? (
                <video
                  key={`${openIndex}-${slideIndex}`}
                  ref={videoRef}
                  src={currentMedia}
                  autoPlay
                  muted={muted}
                  playsInline
                  onTimeUpdate={(e) => {
                    const v = e.currentTarget;
                    if (v.duration > 0) setProgress(Math.min(v.currentTime / v.duration, 1));
                  }}
                  onEnded={advance}
                  onError={advance}
                  className="absolute inset-0 w-full h-full object-cover bg-black"
                />
              ) : (
                // eslint-disable-next-line @next/next/no-img-element
                <img
                  src={currentMedia}
                  alt={currentName}
                  className="absolute inset-0 w-full h-full object-cover bg-black"
                />
              )
            ) : (
              <div className="absolute inset-0 flex items-center justify-center">
                <div className="relative w-32 h-32 sm:w-36 sm:h-36 rounded-2xl
                                overflow-hidden bg-white ring-1 ring-white/20">
                  <Image
                    src={logoVariant(current.logo_url, 144, 144)}
                    alt={currentName}
                    fill
                    sizes="144px"
                    unoptimized
                    className="object-contain p-3"
                  />
                </div>
              </div>
            )}

            {/* ── أشرطة التقدّم (شرائح المتجر الحالي) ── */}
            <div className="absolute top-3 left-3 right-3 z-30 flex gap-1">
              {Array.from({ length: slideCount }).map((_, i) => (
                <div
                  key={i}
                  className="flex-1 h-[2px] bg-white/25 overflow-hidden rounded-full"
                >
                  <div
                    className="h-full bg-white"
                    style={{
                      width:
                        i < slideIndex ? '100%'
                        : i === slideIndex ? `${progress * 100}%`
                        : '0%',
                      transition: i === slideIndex ? 'none' : 'width 200ms',
                    }}
                  />
                </div>
              ))}
            </div>

            {/* إغلاق */}
            <button
              onClick={close}
              aria-label={t('story_close')}
              className="absolute top-6 right-4 z-40 p-2 rounded-full
                         text-white/85 hover:text-white hover:bg-white/10 transition-colors"
            >
              <X className="w-5 h-5" />
            </button>

            {/* كتم/تشغيل الصوت — يظهر للفيديو فقط */}
            {currentIsVideo && (
              <button
                onClick={(e) => { e.stopPropagation(); setMuted((m) => !m); }}
                aria-label={muted ? 'تشغيل الصوت' : 'كتم الصوت'}
                className="absolute top-6 left-4 z-40 p-2 rounded-full
                           text-white/85 hover:text-white bg-black/30 hover:bg-black/50
                           transition-colors"
              >
                {muted ? <VolumeX className="w-5 h-5" /> : <Volume2 className="w-5 h-5" />}
              </button>
            )}

            {/* مناطق النقر: الكل = تقدّم، الحافة المقابلة = رجوع (مع مراعاة RTL) */}
            <button
              type="button"
              onClick={() => { isRTL ? back() : advance(); }}
              aria-label={t('story_next')}
              className="absolute inset-0 z-10 cursor-pointer"
            />
            <button
              type="button"
              onClick={(e) => { e.stopPropagation(); isRTL ? advance() : back(); }}
              aria-label={t('story_prev')}
              className={`absolute top-12 bottom-28 w-[15%] z-20 cursor-pointer ${
                isRTL ? 'right-0' : 'left-0'
              }`}
            />

            {/* ── المحتوى السفلي فوق تدرّج داكن ── */}
            <div className="pointer-events-none absolute inset-x-0 bottom-0 z-20
                            px-8 pt-20 pb-8 flex flex-col items-center text-center text-white
                            bg-gradient-to-t from-black/85 via-black/45 to-transparent">
              <h3 className="text-2xl font-bold tracking-tight mb-1 drop-shadow">
                {currentName}
              </h3>
              {currentOffer && (
                <p className="text-[11px] text-white/70 tracking-[0.25em] uppercase mb-5">
                  Exclusive · {currentOffer}
                </p>
              )}
              {cta}
            </div>
          </div>
        </div>
      )}
    </>
  );
}
