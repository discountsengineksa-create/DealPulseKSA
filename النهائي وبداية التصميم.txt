// ── FETCH ──
async function loadData() {
    try {
        let url;
        if (activeSearch.length >= 2) {
            url = `${API_BASE}/coupons/search?q=${encodeURIComponent(activeSearch)}&limit=50`;
        } else {
            url = `${API_BASE}/coupons/?limit=50`;
        }
        const res = await fetch(url, {
            mode: 'cors',
            headers: { 'Accept': 'application/json' }
        });

        if (!res.ok) throw new Error(`HTTP ${res.status}`);
        const payload = await res.json();
        const items = payload.results || [];

        ALL = items.map(item => {
            let displayName = item.store_id || 'متجر غير معروف';
            if (item.is_trending === 'ترند 🔥') {
                displayName = `🔥 ${item.store_id}`;
            }
            return {
                id:       item.id,
                store_id: displayName,
                name_en:  item.name_en || '',
                emoji:    '🏷️',
                cat:      item.store_bio || '',
                coupon:   item.public_coupon || '',
                disc:     item.discount_value || '',
                link:     item.affiliate_link || '#',
                bio:      item.store_bio || '',
                extra:    item.extra_offer || ''
            };
        });

        applyFilters();
        renderStats();
        renderFeatured();
        renderCats();

    } catch (err) {
        console.error('loadData:', err);
        toast('❌ فشل الاتصال بالسيرفر');
    }
}

    const s = activeSearch;
    filtered = ALL.filter(item => {
        const inCat    = activeCat === 'all' || item.cat === activeCat;
        const inSearch = !s
            || item.store_id.toLowerCase().includes(s)
            || item.name_en.toLowerCase().includes(s)
            || item.coupon.toLowerCase().includes(s);
        return inCat && inSearch;
    });
    renderStores();
    document.getElementById('storeCnt').textContent = filtered.length;
}

function renderCats() {

            <span>${CAT_ICONS[c] || '🏷️'}</span>

function renderStores() {
    const grid  = document.getElementById('storesGrid');
    const empty = document.getElementById('emptyState');

    if (!filtered.length) {
        grid.innerHTML = '';
        empty.classList.add('on');
        return;
    }
    empty.classList.remove('on');

    grid.innerHTML = filtered.map((s, i) => `
        <div class="card glass fade-up" style="animation-delay:${i * .05}s"
             onclick="openStore(${s.id})">
          <div class="card-top">
            <div class="c-ava">${s.emoji}</div>
            ${s.trending ? '<span class="fire">🔥</span>' : ''}
          </div>
          <div class="c-name">${s.store_id}</div>
          <div class="disc-badge">🏷️ ${s.disc}</div>
          ${s.coupon
            ? `<div class="cpn-prev"><span class="cpn-txt">${s.coupon}</span></div>`
            : ''}
          <div class="c-foot">
            <span>${favs.includes(s.id) ? '❤️' : '🤍'}</span>
            <span class="go-lbl">نسخ ←</span>
          </div>
        </div>`).join('');
}

// ── MODAL ──
function openStore(id) {
    const s = ALL.find(x => x.id === id);
    if (!s) return;
    haptic('medium');
    const isFav = favs.includes(id);
    document.getElementById('mBody').innerHTML = `
        <div class="m-hero">
          <div class="m-ava">${s.emoji}</div>
          <div class="m-info"><h2>${s.store_id}</h2><p>${s.cat}</p></div>
        </div>
        <div class="m-desc glass">${s.bio}</div>
        <div class="cpn-lbl">🎟️ كود الخصم</div>
        <div class="cpn-box" onclick="copyCoupon('${s.coupon}', ${id})">
          <span class="big-cpn">${s.coupon || '—'}</span>
          <button class="copy-btn">نسخ</button>
          <div class="copy-ok" id="cok_${id}">✅ تم النسخ!</div>
        </div>
        <div class="m-actions">
          <button class="btn-p" onclick="goToStore('${s.link}')">🛍️ انتقل للمتجر</button>
          <button class="btn-s glass" onclick="toggleFav(${id})">
            ${isFav ? '💔 إزالة من المفضلة' : '❤️ أضف للمفضلة'}
          </button>
        </div>`;
    document.getElementById('overlay').classList.add('on');
}

function copyCoupon(code, id) {
    if (navigator.clipboard) {
        navigator.clipboard.writeText(code);
    } else {
        const t = document.createElement('textarea');
        t.value = code;
        document.body.appendChild(t);
        t.select();
        document.execCommand('copy');
        document.body.removeChild(t);
    }
    haptic('success');
    document.getElementById(`cok_${id}`)?.classList.add('on');
    toast('✅ تم النسخ: ' + code);
    setTimeout(() => document.getElementById(`cok_${id}`)?.classList.remove('on'), 2000);
}

function goToStore(link) {
    haptic();
    if (tg) tg.openLink(link);
    else window.open(link, '_blank');
}

function toggleFav(id) {
    haptic();
    const i = favs.indexOf(id);
    if (i === -1) favs.push(id); else favs.splice(i, 1);
    localStorage.setItem('dp_favs', JSON.stringify(favs));
    closeModal();
    renderStores();
    toast(i === -1 ? '❤️ أُضيف للمفضلة' : '💔 حُذف من المفضلة');
}

// ── CONTROLS ──
function closeModal()      { document.getElementById('overlay').classList.remove('on'); }
function onOverlayClick(e) { if (e.target.id === 'overlay') closeModal(); }
function onCat(c)          { haptic(); activeCat = c; applyFilters(); renderCats(); }
function onSearch(v)       { activeSearch = v.toLowerCase().trim(); loadData(); }
function switchNav(t) {
    haptic();
    document.querySelectorAll('.nitem').forEach(n => n.classList.remove('on'));
    document.getElementById(`nav-${t}`)?.classList.add('on');
}
async function refreshData() { toast('🔄 جاري التحديث...'); await loadData(); }

// ── TOAST ──
function toast(msg, dur = 2400) {
    const el = document.getElementById('toast');
    el.textContent = msg;
    el.classList.add('on');
    setTimeout(() => el.classList.remove('on'), dur);
}
